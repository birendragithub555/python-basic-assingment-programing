def add_new(a,b):
    l=[]
    if a==[]:
        print("no list selected")
    else:
        for i in range(1,len(a)):
            l.append(a[i])
        l.append(b)
        return l
add_new([1,2,3,4,5],6)
add_new([],6)

def get_budgets(a):
    d=0
    for i in a:
        d=d+i["budget"]
    return d

  { "name": "John", "age": 21, "budget": 23000 },
  { "name": "Steve",  "age": 32, "budget": 40000 },
  { "name": "Martin",  "age": 16, "budget": 2700 }])
